DROP PROCEDURE IF EXISTS "OrigamIdentityGrantCleanup";

CREATE PROCEDURE "OrigamIdentityGrantCleanup"()
LANGUAGE plpgsql
AS $$
DECLARE
    current_datetime TIMESTAMP := now();
BEGIN
    LOOP
        DELETE FROM "OrigamIdentityGrant"
        WHERE "Expiration" < current_datetime
        LIMIT 1000;
        EXIT WHEN NOT FOUND;
    END LOOP;
END;
$$;
